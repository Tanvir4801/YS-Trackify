```jsx
function AttendanceHeatmap({ labourId }) {
  // Show last 30 days as a grid of colored squares
  // Each square = one day
  // Color based on attendance status
  // Hover shows date and status tooltip
  const [data, setData] = useState({});
  
  useEffect(() => {
    // Fetch last 30 days attendance for this labour
    const days = [];
    for (let i = 29; i >= 0; i--) {
      const d = new Date();
      d.setDate(d.getDate() - i);
      days.push(d.toISOString().split('T')[0]);
    }
    // Query attendance collection
    // Map labourId → status for each date
  }, [labourId]);

  return (
    <div className="grid grid-cols-10 gap-1">
      {days.map(day => (
        <div
          key={day}
          title={`${day}: ${data[day] || 'No record'}`}
          className={`w-6 h-6 rounded-sm cursor-pointer ${
            data[day] === 'present' ? 'bg-green-500' :
            data[day] === 'absent'  ? 'bg-red-400'   :
            data[day] === 'half'    ? 'bg-orange-400' :
            'bg-gray-200'
          }`}
        />
      ))}
    </div>
  );
}
```

### CSV export utility
```javascript
// lib/utils/exportCSV.js
export const exportCSV = (data, filename, columns) => {
  // columns: [{ key: 'name', label: 'Labour Name' }, ...]
  const BOM = '\uFEFF';
  const headers = columns.map(c => c.label);
  const rows = data.map(item =>
    columns.map(c => {
      const val = item[c.key] ?? '';
      // Format special types
      if (val instanceof Date) {
        return val.toLocaleDateString('en-IN');
      }
      if (typeof val === 'number' && c.currency) {
        return `₹${val.toLocaleString('en-IN')}`;
      }
      return String(val).replace(/"/g, '""');
    })
  );
  const csv = BOM + [headers, ...rows]
    .map(r => r.map(v => `"${v}"`).join(','))
    .join('\n');
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${filename}_${new Date().toISOString().split('T')[0]}.csv`;
  a.click();
  URL.revokeObjectURL(url);
};
```

```javascript
// lib/services/activityService.js
export const logActivity = async (action, details, userId) => {
  await addDoc(collection(db, 'activity_log'), {
    action,      // 'labour_added' | 'attendance_marked' etc
    details,     // human readable string
    userId,
    timestamp: serverTimestamp(),
  });
};

export const getRecentActivity = async (limit = 50) => {
  const snap = await getDocs(query(
    collection(db, 'activity_log'),
    orderBy('timestamp', 'desc'),
    limit(limit)
  ));
  return snap.docs.map(d => ({
    id: d.id, ...d.data(),
    timestamp: d.data().timestamp?.toDate?.() || null,
  }));
};
```
