'use client';
import { db } from '../firebase';
import {
  collection, getDocs, addDoc, updateDoc, deleteDoc,
  doc, serverTimestamp, query, where, onSnapshot
} from 'firebase/firestore';

// Fetch labours — query BOTH supervisorId AND contractorId
// because Flutter app stores supervisor UID in both fields
export const getLabours = async (supervisorId = null) => {
  try {
    let results = [];

    if (supervisorId) {
      // Query by supervisorId
      const snap1 = await getDocs(query(
        collection(db, 'labours'),
        where('supervisorId', '==', supervisorId),
        where('isActive', '==', true)
      ));
      // Query by contractorId (same UID stored in different field)
      const snap2 = await getDocs(query(
        collection(db, 'labours'),
        where('contractorId', '==', supervisorId),
        where('isActive', '==', true)
      ));

      // Merge and deduplicate by doc ID
      const map = new Map();
      [...snap1.docs, ...snap2.docs].forEach(d => {
        map.set(d.id, {
          id: d.id,
          ...d.data(),
          joiningDate: d.data().joiningDate?.toDate?.() || null,
          syncedAt: d.data().syncedAt?.toDate?.() || null,
          // Handle missing overtime fields gracefully
          overtimeWagePerHour: d.data().overtimeWagePerHour || 0,
          defaultOvertimeHours: d.data().defaultOvertimeHours || 0,
        });
      });
      results = Array.from(map.values());
    } else {
      // Admin — fetch all labours
      const snap = await getDocs(query(
        collection(db, 'labours'),
        where('isActive', '==', true)
      ));
      results = snap.docs.map(d => ({
        id: d.id,
        ...d.data(),
        joiningDate: d.data().joiningDate?.toDate?.() || null,
        overtimeWagePerHour: d.data().overtimeWagePerHour || 0,
        defaultOvertimeHours: d.data().defaultOvertimeHours || 0,
      }));
    }

    return results;
  } catch (error) {
    console.error('getLabours error:', error);
    throw error;
  }
};

// Real-time stream — also queries both fields
export const subscribeLabours = (supervisorId, callback) => {
  const results = new Map();

  const update = () => {
    const merged = Array.from(results.values())
      .sort((a, b) => (a.name || '').localeCompare(b.name || ''));
    callback(merged);
  };

  const constraints = [where('isActive', '==', true)];
  if (supervisorId) {
    constraints.push(where('supervisorId', '==', supervisorId));
  }

  const unsub1 = onSnapshot(
    query(collection(db, 'labours'), ...constraints),
    (snap) => {
      snap.docs.forEach(d => {
        results.set(d.id, {
          id: d.id, ...d.data(),
          joiningDate: d.data().joiningDate?.toDate?.() || null,
          overtimeWagePerHour: d.data().overtimeWagePerHour || 0,
        });
      });
      update();
    }
  );

  return unsub1;
};

export const addLabour = async (data) => {
  const ref = await addDoc(collection(db, 'labours'), {
    ...data,
    overtimeWagePerHour: data.overtimeWagePerHour || 0,
    defaultOvertimeHours: data.defaultOvertimeHours || 0,
    isActive: true,
    isSynced: true,
    syncedAt: serverTimestamp(),
    createdAt: serverTimestamp(),
  });
  await updateDoc(ref, { id: ref.id });
  return ref.id;
};

export const updateLabour = async (id, data) => {
  await updateDoc(doc(db, 'labours', id), {
    ...data,
    updatedAt: serverTimestamp(),
  });
};

export const deleteLabour = async (id) => {
  await updateDoc(doc(db, 'labours', id), {
    isActive: false,
    deletedAt: serverTimestamp(),
  });

'use client';
import { db } from '../firebase';
import {
  collection, getDocs, addDoc, updateDoc,
  doc, serverTimestamp, query, where,
  onSnapshot, writeBatch, Timestamp
} from 'firebase/firestore';

// Helper: today as "YYYY-MM-DD"
export const getTodayString = () => {
  const now = new Date();
  return `${now.getFullYear()}-${
    String(now.getMonth()+1).padStart(2,'0')}-${
    String(now.getDate()).padStart(2,'0')}`;
};

// Get attendance for a specific date
export const getAttendanceByDate = async (date) => {
  const snap = await getDocs(query(
    collection(db, 'attendance'),
    where('date', '==', date)
  ));
  return snap.docs.map(d => ({ id: d.id, ...d.data() }));
};

// Get attendance for date range — for reports
export const getAttendanceByMonth = async (month, year, supervisorId) => {
  const monthStr = `${year}-${String(month).padStart(2, '0')}`;
  const constraints = [
    where('date', '>=', `${monthStr}-01`),
    where('date', '<=', `${monthStr}-31`),
  ];
  if (supervisorId) {
    constraints.push(where('supervisorId', '==', supervisorId));
  }
  const snap = await getDocs(query(
    collection(db, 'attendance'), ...constraints
  ));
  return snap.docs.map(d => ({ id: d.id, ...d.data() }));
};

// Update attendance status (present/absent/half)
export const updateAttendanceStatus = async (id, status) => {
  await updateDoc(doc(db, 'attendance', id), {
    status,
    updatedAt: serverTimestamp(),
  });
};

// Update overtime hours
export const updateOvertimeHours = async (id, overtimeHours) => {
  await updateDoc(doc(db, 'attendance', id), {
    overtimeHours: parseFloat(overtimeHours) || 0,
    updatedAt: serverTimestamp(),
  });
};

// Bulk mark attendance for all labours of a supervisor
export const bulkMarkAttendance = async (
  supervisorId, date, status, labours
) => {
  const batch = writeBatch(db);

  for (const labour of labours) {
    // Check if record already exists
    const existing = await getDocs(query(
      collection(db, 'attendance'),
      where('labourId', '==', labour.id),
      where('date', '==', date),
      where('supervisorId', '==', supervisorId)
    ));

    if (existing.docs.length > 0) {
      batch.update(existing.docs[0].ref, {
        status,
        updatedAt: serverTimestamp(),
      });
    } else {
      const ref = doc(collection(db, 'attendance'));
      batch.set(ref, {
        id: ref.id,
        labourId: labour.id,
        supervisorId,
        date,
        status,
        overtimeHours: 0,
        isSynced: true,
        syncedAt: serverTimestamp(),
      });
    }
  }
  await batch.commit();
};

// Real-time stream for today attendance — for dashboard
export const subscribeTodayAttendance = (callback) => {
  const today = getTodayString();
  return onSnapshot(
    query(collection(db, 'attendance'), where('date', '==', today)),
    (snap) => {
      let present = 0, absent = 0, half = 0;
      snap.docs.forEach(d => {
        const s = d.data().status;
        if (s === 'present') present++;
        else if (s === 'absent') absent++;
        else if (s === 'half') half++;
      });
      callback({
        present, absent, half,
        total: snap.size,
        docs: snap.docs.map(d => ({ id: d.id, ...d.data() }))
      });
    }
  );
};

// Export attendance as CSV
export const exportAttendanceCSV = (data, labourMap, filename) => {
  const headers = [
    'Labour Name', 'Phone', 'Date',
    'Status', 'Overtime Hours', 'Supervisor ID'
  ];
  const rows = data.map(record => {
    const labour = labourMap[record.labourId] || {};
    return [
      labour.name || record.labourId,
      labour.phone || '',
      record.date,
      record.status,
      record.overtimeHours || 0,
      record.supervisorId,
    ];
  });

  const BOM = '\uFEFF';
  const csv = BOM + [headers, ...rows]
    .map(row => row.map(v => `"${v}"`).join(','))
    .join('\n');

  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.href = url;
  link.download = `${filename || 'attendance'}_${getTodayString()}.csv`;
  link.click();
  URL.revokeObjectURL(url);
};

'use client';
import { db } from '../firebase';
import { collection, getDocs, query, where, Timestamp } from 'firebase/firestore';

export const getMonthlySalaryReport = async (month, year, supervisorId) => {
  const monthStr = `${year}-${String(month).padStart(2, '0')}`;

  // Get labours — query both fields
  const [snap1, snap2] = await Promise.all([
    getDocs(query(
      collection(db, 'labours'),
      where('supervisorId', '==', supervisorId),
      where('isActive', '==', true)
    )),
    getDocs(query(
      collection(db, 'labours'),
      where('contractorId', '==', supervisorId),
      where('isActive', '==', true)
    )),
  ]);

  const labourMap = new Map();
  [...snap1.docs, ...snap2.docs].forEach(d => {
    labourMap.set(d.id, { id: d.id, ...d.data() });
  });
  const labours = Array.from(labourMap.values());

  const report = [];

  for (const labour of labours) {
    // Attendance this month — date field is "YYYY-MM-DD" string
    const attSnap = await getDocs(query(
      collection(db, 'attendance'),
      where('labourId', '==', labour.id),
      where('date', '>=', `${monthStr}-01`),
      where('date', '<=', `${monthStr}-31`)
    ));

    const records = attSnap.docs.map(d => d.data());
    const daysPresent = records.filter(r => r.status === 'present').length;
    const daysHalf    = records.filter(r => r.status === 'half').length;
    const daysAbsent  = records.filter(r => r.status === 'absent').length;
    const totalOTHours = records.reduce(
      (s, r) => s + (r.overtimeHours || 0), 0
    );
    const totalDays   = daysPresent + (daysHalf * 0.5);
    const dailyWage   = labour.dailyWage || 0;
    const otRate      = labour.overtimeWagePerHour || 0;
    const grossSalary = (totalDays * dailyWage) + (totalOTHours * otRate);

    // Advances this month
    const startDate = new Date(year, month - 1, 1);
    const endDate   = new Date(year, month, 0, 23, 59, 59);
    const advSnap   = await getDocs(query(
      collection(db, 'payments'),
      where('labourId', '==', labour.id),
      where('type', '==', 'advance'),
      where('date', '>=', Timestamp.fromDate(startDate)),
      where('date', '<=', Timestamp.fromDate(endDate))
    ));

    const totalAdvances = advSnap.docs
      .reduce((s, d) => s + (d.data().amount || 0), 0);

    report.push({
      labourId:      labour.id,
      labourName:    labour.name,
      phone:         labour.phone,
      dailyWage,
      otRate,
      daysPresent,
      daysHalf,
      daysAbsent,
      totalOTHours,
      totalDays,
      grossSalary,
      totalAdvances,
      netPayable:    grossSalary - totalAdvances,
    });
  }

  // Sort by name
  return report.sort((a, b) => a.labourName.localeCompare(b.labourName));
};

// Export salary report as CSV
export const exportSalaryCSV = (report, month, year) => {
  const monthName = new Date(year, month-1).toLocaleString('default', {month: 'long'});
  const headers = [
    'Name', 'Phone', 'Daily Wage', 'OT Rate/hr',
    'Days Present', 'Half Days', 'Days Absent',
    'OT Hours', 'Total Days', 'Gross Salary',
    'Advances', 'Net Payable'
  ];
  const rows = report.map(r => [
    r.labourName, r.phone, r.dailyWage, r.otRate,
    r.daysPresent, r.daysHalf, r.daysAbsent,
    r.totalOTHours, r.totalDays, r.grossSalary,
    r.totalAdvances, r.netPayable
  ]);

  const BOM = '\uFEFF';
  const csv = BOM + [headers, ...rows]
    .map(row => row.map(v => `"${v}"`).join(','))
    .join('\n');

  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
  const url  = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.href     = url;
  link.download = `Salary_Report_${monthName}_${year}.csv`;
  link.click();
  URL.revokeObjectURL(url);
};

  const [stats, setStats] = useState({
    totalLabours: 0,
    totalSupervisors: 0,
    todayPresent: 0,
    todayAbsent: 0,
    todayHalf: 0,
    monthlyPayments: 0,
  });
  const [recentAttendance, setRecentAttendance] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const unsubs = [];

    // Real-time: total active labours
    unsubs.push(onSnapshot(
      query(collection(db,'labours'), where('isActive','==',true)),
