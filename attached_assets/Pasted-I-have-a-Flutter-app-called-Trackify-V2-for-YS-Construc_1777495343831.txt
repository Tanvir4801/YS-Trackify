### Update lib/services/auth_service.dart

Add this method:
```dart
// Labour login by phone number only
// No OTP needed — phone is the unique identifier
Future<AuthResult> labourLoginByPhone(String phone) async {
  try {
    final phoneClean = phone.trim().replaceAll('+91', '');
    
    if (phoneClean.length != 10) {
      return AuthResult.error('Enter valid 10 digit number');
    }

    // Step 1: Check labours collection by phone
    final labourSnap = await _db
        .collection('labours')
        .where('phone', isEqualTo: phoneClean)
        .where('isActive', isEqualTo: true)
        .limit(1)
        .get();

    if (labourSnap.docs.isEmpty) {
      return AuthResult.error(
        'Mobile number not registered.\nContact your supervisor.'
      );
    }

    final labourData = labourSnap.docs.first.data();
    final labourId = labourSnap.docs.first.id;
    final labourName = labourData['name'] ?? 'Labour';
    final supervisorId = labourData['supervisorId'] ?? '';

    // Step 2: Check if Firebase Auth user exists for this phone
    // Try signing in with phone via Firebase Auth
    await _auth.verifyPhoneNumber(
      phoneNumber: '+91$phoneClean',
      timeout: const Duration(seconds: 60),
      verificationCompleted: (credential) async {
        await _auth.signInWithCredential(credential);
      },
      verificationFailed: (e) {
        debugPrint('Labour auth error: ${e.code}');
      },
      codeSent: (verificationId, _) {
        // Store for OTP verification
        _labourVerificationId = verificationId;
      },
      codeAutoRetrievalTimeout: (_) {},
    );

    // Step 3: Check/create users doc for this labour
    final existingUser = await _db
        .collection('users')
        .where('phone', isEqualTo: phoneClean)
        .where('role', isEqualTo: 'labour')
        .limit(1)
        .get();

    if (existingUser.docs.isEmpty) {
      // Create user doc for labour
      final newRef = _db.collection('users').doc();
      await newRef.set({
        'uid': newRef.id,
        'phone': phoneClean,
        'name': labourName,
        'role': 'labour',
        'isActive': true,
        'supervisorId': supervisorId,
        'labourId': labourId,
        'createdAt': FieldValue.serverTimestamp(),
      });
    }

    // Cache labour session locally
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString('role', 'labour');
    await prefs.setString('name', labourName);
    await prefs.setString('phone', phoneClean);
    await prefs.setString('labourId', labourId);
    await prefs.setString('supervisorId', supervisorId);
    await prefs.setBool('isLoggedIn', true);

    return AuthResult.success(
      uid: labourId,
      role: 'labour',
      name: labourName,
    );
  } catch (e) {
    debugPrint('Labour login error: $e');
    return AuthResult.error('Login failed. Try again.');
  }
}

String? _labourVerificationId;
```

// Labour login screen — simplified
class LabourLoginScreen extends StatefulWidget {
  const LabourLoginScreen({super.key});
  @override
  State<LabourLoginScreen> createState() => _LabourLoginScreenState();
}

class _LabourLoginScreenState extends State<LabourLoginScreen> {
  final _phoneController = TextEditingController();
  final _authService = AuthService();
  bool _isLoading = false;
  String? _error;

  Future<void> _login() async {
    final phone = _phoneController.text.trim();
    if (phone.length != 10) {
      setState(() => _error = 'Enter valid 10 digit mobile number');
      return;
    }

    setState(() { _isLoading = true; _error = null; });

    final result = await _authService.labourLoginByPhone(phone);

    if (!mounted) return;

    if (result.success) {
      Navigator.of(context).pushNamedAndRemoveUntil(
        '/labour-home', (route) => false,
      );
    } else {
      setState(() {
        _error = result.errorMessage;
        _isLoading = false;
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    // Keep EXACT existing UI — only change form fields
    return Scaffold(
      appBar: AppBar(title: const Text('Labour Mode')),
      body: Padding(
        padding: const EdgeInsets.all(24),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            const Text('Labour Login',
              style: TextStyle(
                fontSize: 22, fontWeight: FontWeight.bold)),
            const SizedBox(height: 8),
            const Text('Login with your registered mobile number.',
              style: TextStyle(color: Colors.grey)),
            const SizedBox(height: 32),

            // Only mobile number field
            TextField(
              controller: _phoneController,
              keyboardType: TextInputType.phone,
              maxLength: 10,

                hintText: 'Mobile Number',

## FIX 3 — Wage Snapshot Calculation Fix

PROBLEM: Dashboard shows "Rs 0" for Today/Week/Month wages
even when attendance is marked.

FIX: Update DashboardProvider wage calculation:

```dart
Future<double> calculateTodayWages() async {
  final today = _todayString();
  final snap = await _db
      .collection('attendance')
      .where('supervisorId', isEqualTo: supervisorId)
      .where('date', isEqualTo: today)
      .get();

  double total = 0;
  for (var doc in snap.docs) {
    final labourId = doc['labourId'] as String;
    final status = doc['status'] as String;
    final overtimeHours = 
      (doc['overtimeHours'] as num?)?.toDouble() ?? 0;

    // Get labour wage
    final labourSnap = await _db
        .collection('labours')
        .where('id', isEqualTo: labourId)
        .limit(1)
        .get();

    if (labourSnap.docs.isNotEmpty) {
      final dailyWage = 
        (labourSnap.docs.first['dailyWage'] as num).toDouble();
      final overtimeRate = 
        (labourSnap.docs.first['overtimeWagePerHour'] 
          as num?)?.toDouble() ?? 0;

      if (status == 'present') {
        total += dailyWage + (overtimeHours * overtimeRate);
      } else if (status == 'half') {
        total += (dailyWage / 2) + (overtimeHours * overtimeRate);
      }
      // absent = 0
    }
  }
  return total;
}
```

## FIX 6 — Labour Home Screen: Show Own Attendance

When labour logs in with their phone number, their home screen 
should show their own attendance history fetched from Firebase.

```dart
// In LabourHomeScreen initState:
Future<void> _loadMyAttendance() async {
  final prefs = await SharedPreferences.getInstance();
  final labourId = prefs.getString('labourId') ?? '';
  final phone = prefs.getString('phone') ?? '';

  if (labourId.isEmpty) return;

  // Fetch last 30 days attendance
  final now = DateTime.now();
  final thirtyDaysAgo = now.subtract(const Duration(days: 30));
  final fromStr = 
    '${thirtyDaysAgo.year}-${thirtyDaysAgo.month.toString().padLeft(2,'0')}-01';

  final snap = await FirebaseFirestore.instance
      .collection('attendance')
      .where('labourId', isEqualTo: labourId)
      .where('date', isGreaterThanOrEqualTo: fromStr)
      .orderBy('date', descending: true)
      .get();

  setState(() {
    myAttendance = snap.docs.map((d) => d.data()).toList();
  });
}
```
