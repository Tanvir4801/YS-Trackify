import 'dart:async';
import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:firebase_auth/firebase_auth.dart';
import 'package:flutter/foundation.dart';

class DashboardProvider extends ChangeNotifier {
  final _db   = FirebaseFirestore.instance;
  final _auth = FirebaseAuth.instance;

  // ── State ──────────────────────────────────────────────
  String  contractorName   = '';
  int     totalLabours     = 0;
  int     presentToday     = 0;
  int     absentToday      = 0;
  int     halfToday        = 0;
  double  todayWages       = 0;
  double  weekWages        = 0;
  double  monthWages       = 0;
  double  totalAdvances    = 0;
  bool    isLoading        = true;
  String? error;

  // Live attendance for today
  Map<String, Map<String, dynamic>> _todayAttMap  = {};
  Map<String, Map<String, dynamic>> _labourWageMap = {};

  // Subscriptions
  final List<StreamSubscription> _subs = [];

  String get uid => _auth.currentUser?.uid ?? '';

  // ── Initialize — call once from initState ──────────────
  Future<void> initialize() async {
    if (uid.isEmpty) return;
    isLoading = true;
    notifyListeners();

    await _loadContractorName();
    await _loadLabours();          // one-time then stream
    _startTodayAttendanceStream(); // real-time
    _startWeekStream();            // real-time
    _startMonthStream();           // real-time
    _startAdvanceStream();         // real-time

    isLoading = false;
    notifyListeners();
  }

  // ── Contractor name ─────────────────────────────────────
  Future<void> _loadContractorName() async {
    try {
      final userDoc = await _db.collection('users').doc(uid).get();
      final contractorId =
          userDoc.data()?['contractorId'] as String? ?? uid;

      DocumentSnapshot? cDoc;
      try {
        cDoc = await _db.collection('contractors').doc(contractorId).get();
      } catch (_) {}

      if (cDoc != null && cDoc.exists) {
        contractorName = cDoc['name'] as String? ?? 'My Company';
      } else {
        contractorName = userDoc.data()?['name'] as String? ?? 'My Company';
      }
      notifyListeners();
    } catch (e) {
      contractorName = 'My Company';
    }
  }

  // ── Labours — stream for live count and wage map ────────
  Future<void> _loadLabours() async {
    try {
      // Fetch once to build wageMap immediately
      final s1 = await _db.collection('labours')
          .where('supervisorId', isEqualTo: uid)
          .where('isActive', isEqualTo: true).get();
      final s2 = await _db.collection('labours')
          .where('contractorId', isEqualTo: uid)
          .where('isActive', isEqualTo: true).get();

      final map = <String, Map<String, dynamic>>{};
      for (var d in [...s1.docs, ...s2.docs]) {
        map[d.id] = d.data();
      }
      _labourWageMap = map;
      totalLabours   = map.length;
      notifyListeners();
    } catch (e) {
      debugPrint('_loadLabours error: $e');
    }

    // Then subscribe for live updates
    final sub1 = _db.collection('labours')
        .where('supervisorId', isEqualTo: uid)
        .where('isActive', isEqualTo: true)
        .snapshots()
        .listen((snap) {
      for (var d in snap.docs) _labourWageMap[d.id] = d.data();
      // Remove deleted
      final ids = snap.docs.map((d) => d.id).toSet();
      _labourWageMap.removeWhere((k, _) => !ids.contains(k));
      totalLabours = _labourWageMap.length;
      _recalculateWages();
    }, onError: (e) => debugPrint('labours stream error: $e'));
    _subs.add(sub1);

    final sub2 = _db.collection('labours')
        .where('contractorId', isEqualTo: uid)
        .where('isActive', isEqualTo: true)
        .snapshots()
        .listen((snap) {
      for (var d in snap.docs) {
        _labourWageMap[d.id] = d.data();
      }
      totalLabours = _labourWageMap.length;
      _recalculateWages();
    }, onError: (e) => debugPrint('labours contractorId stream error: $e'));
    _subs.add(sub2);
  }

  // ── Today attendance stream ─────────────────────────────
  void _startTodayAttendanceStream() {
    final today = _dateStr(DateTime.now());

    final sub = _db.collection('attendance')
        .where('supervisorId', isEqualTo: uid)
        .where('date', isEqualTo: today)
        .snapshots()
        .listen((snap) {
      _todayAttMap = {
        for (var d in snap.docs) d.data()['labourId'] as String: d.data()
      };
      _updateTodayCounts();
      _recalculateWages();
    }, onError: (e) => debugPrint('today att stream: $e'));
    _subs.add(sub);

    // Also listen to contractorId field
    final sub2 = _db.collection('attendance')
        .where('contractorId', isEqualTo: uid)
        .where('date', isEqualTo: today)
        .snapshots()
        .listen((snap) {
      for (var d in snap.docs) {
        final lid = d.data()['labourId'] as String? ?? '';
        if (lid.isNotEmpty) _todayAttMap[lid] = d.data();
      }
      _updateTodayCounts();
      _recalculateWages();
    }, onError: (e) => debugPrint('today att contractorId stream: $e'));
    _subs.add(sub2);
  }

  void _updateTodayCounts() {
    presentToday = _todayAttMap.values
        .where((a) => a['status'] == 'present').length;
    absentToday  = _todayAttMap.values
        .where((a) => a['status'] == 'absent').length;
    halfToday    = _todayAttMap.values
        .where((a) => a['status'] == 'half').length;
    notifyListeners();
  }

  // ── Week stream ─────────────────────────────────────────
  void _startWeekStream() {
    final now   = DateTime.now();
    final start = now.subtract(const Duration(days: 6));

    final sub = _db.collection('attendance')
        .where('supervisorId', isEqualTo: uid)
        .where('date', isGreaterThanOrEqualTo: _dateStr(start))
        .where('date', isLessThanOrEqualTo:    _dateStr(now))
        .snapshots()
        .listen((snap) {
      double total = 0;
      for (var d in snap.docs) {
        total += _wageForRecord(d.data());
      }
      weekWages = total;
      notifyListeners();
    }, onError: (e) => debugPrint('week stream: $e'));
    _subs.add(sub);
  }

  // ── Month stream ────────────────────────────────────────
  void _startMonthStream() {
    final now   = DateTime.now();
    final start = DateTime(now.year, now.month, 1);

    final sub = _db.collection('attendance')
        .where('supervisorId', isEqualTo: uid)
        .where('date', isGreaterThanOrEqualTo: _dateStr(start))
        .where('date', isLessThanOrEqualTo:    _dateStr(now))
        .snapshots()
        .listen((snap) {
      double total = 0;
      for (var d in snap.docs) {
        total += _wageForRecord(d.data());
      }
      monthWages = total;
      notifyListeners();
    }, onError: (e) => debugPrint('month stream: $e'));
    _subs.add(sub);
  }

  // ── Advance stream ──────────────────────────────────────
  void _startAdvanceStream() {
    final sub = _db.collection('payments')
        .where('supervisorId', isEqualTo: uid)
        .where('type', isEqualTo: 'advance')
        .snapshots()
        .listen((snap) {
      totalAdvances = snap.docs.fold(0.0,
          (s, d) => s + ((d.data()['amount'] as num?)?.toDouble() ?? 0));
      notifyListeners();
    }, onError: (e) => debugPrint('advance stream: $e'));
    _subs.add(sub);
  }

  // ── Wage calculation ────────────────────────────────────
  double _wageForRecord(Map<String, dynamic> att) {
    final labourId = att['labourId'] as String? ?? '';
    final status   = att['status']   as String? ?? '';
    final ot       = (att['overtimeHours'] as num?)?.toDouble() ?? 0;

    // Use snapshot wage if stored, fallback to labourWageMap
    final snapWage = (att['dailyWageSnapshot'] as num?)?.toDouble();
    final labData  = _labourWageMap[labourId];
    final wage = snapWage ?? 
        (labData?['dailyWage'] as num?)?.toDouble() ?? 0;
    final otRate = (labData?['overtimeWagePerHour'] as num?)?.toDouble() ?? 0;

    double pay = 0;
    if (status == 'present') pay = wage + (ot * otRate);
    else if (status == 'half') pay = (wage / 2) + (ot * otRate);
    return pay;
  }

  void _recalculateWages() {
    double today = 0;
    for (final att in _todayAttMap.values) {
      today += _wageForRecord(att);
    }
    todayWages = today;
    notifyListeners();
  }

  // ── Helpers ─────────────────────────────────────────────
  String _dateStr(DateTime d) =>
      '${d.year}-${d.month.toString().padLeft(2,'0')}-'
      '${d.day.toString().padLeft(2,'0')}';

  Future<void> refresh() async {
    _subs.clear();
    await initialize();
  }

  @override
  void dispose() {
    for (var s in _subs) s.cancel();
    super.dispose();
  }
}

// lib/providers/reports_provider.dart

class ReportsProvider extends ChangeNotifier {
  final _db   = FirebaseFirestore.instance;
  final _auth = FirebaseAuth.instance;

  int    selectedMonth = DateTime.now().month;
  int    selectedYear  = DateTime.now().year;
  bool   isLoading     = false;
  String? error;

  List<LabourReport> reports     = [];
  double totalGross              = 0;
  double totalAdvances           = 0;
  double totalNet                = 0;

  final List<StreamSubscription> _subs = [];
  String get uid => _auth.currentUser?.uid ?? '';

  // Call when month changes or on init
  void loadReport({int? month, int? year}) {
    if (month != null) selectedMonth = month;
    if (year  != null) selectedYear  = year;
    _cancelSubs();
    _startReportStream();
    notifyListeners();
  }

  void _startReportStream() {
    isLoading = true;
    notifyListeners();

    final monthStr = '$selectedYear-'
        '${selectedMonth.toString().padLeft(2,'0')}';

    // Stream labours
    final labourSub = _db.collection('labours')
        .where('supervisorId', isEqualTo: uid)
        .where('isActive', isEqualTo: true)
        .snapshots()
        .listen((labourSnap) async {
      final labours = {
        for (var d in labourSnap.docs) d.id: d.data()
      };
      await _buildReports(labours, monthStr);
    }, onError: (e) {
      error = e.toString();
      isLoading = false;
      notifyListeners();
    });
    _subs.add(labourSub);

    // Also listen to attendance for this month — 
    // when anything changes, rebuild report
    final attSub = _db.collection('attendance')
        .where('supervisorId', isEqualTo: uid)
        .where('date', isGreaterThanOrEqualTo: '$monthStr-01')
        .where('date', isLessThanOrEqualTo:    '$monthStr-31')
        .snapshots()
        .listen((_) async {
      // Re-trigger report build when attendance changes
      final labourSnap = await _db.collection('labours')
          .where('supervisorId', isEqualTo: uid)
          .where('isActive', isEqualTo: true).get();
      final labours = {for (var d in labourSnap.docs) d.id: d.data()};
      await _buildReports(labours, monthStr);
    }, onError: (e) => debugPrint('att report stream: $e'));
    _subs.add(attSub);

    // Also listen to payments (advances) for this month
    final startDate = DateTime(selectedYear, selectedMonth, 1);
    final endDate   = DateTime(selectedYear, selectedMonth + 1, 0);
    final paymentSub = _db.collection('payments')
        .where('supervisorId', isEqualTo: uid)
        .where('type', isEqualTo: 'advance')
        .snapshots()
        .listen((_) async {
      final labourSnap = await _db.collection('labours')
          .where('supervisorId', isEqualTo: uid)
          .where('isActive', isEqualTo: true).get();
      final labours = {for (var d in labourSnap.docs) d.id: d.data()};
      await _buildReports(labours, monthStr);
    }, onError: (e) => debugPrint('payment report stream: $e'));
    _subs.add(paymentSub);
  }

  Future<void> _buildReports(
    Map<String, Map<String, dynamic>> labours,
    String monthStr,
  ) async {
    final results = <LabourReport>[];

    for (final entry in labours.entries) {
      final labourId = entry.key;
      final labData  = entry.value;

      // Attendance for this labour this month
      final attSnap = await _db.collection('attendance')
          .where('labourId', isEqualTo: labourId)
          .where('date', isGreaterThanOrEqualTo: '$monthStr-01')
          .where('date', isLessThanOrEqualTo:    '$monthStr-31')
          .get();

      int    daysPresent = 0;
      int    daysHalf    = 0;
      int    daysAbsent  = 0;
      double totalOT     = 0;

      for (var d in attSnap.docs) {
        final status = d.data()['status'] as String? ?? '';
        final ot     = (d.data()['overtimeHours'] as num?)?.toDouble() ?? 0;
        totalOT += ot;
        if (status == 'present')     daysPresent++;
        else if (status == 'half')   daysHalf++;
        else if (status == 'absent') daysAbsent++;
      }

      final totalDays = daysPresent + (daysHalf * 0.5);
      final dailyWage = (labData['dailyWage'] as num?)?.toDouble() ?? 0;
      final otRate    = (labData['overtimeWagePerHour'] as num?)?.toDouble() ?? 0;
      final gross     = (totalDays * dailyWage) + (totalOT * otRate);

      // Advances this month
      final startTs = Timestamp.fromDate(
          DateTime(int.parse(monthStr.split('-')[0]),
                   int.parse(monthStr.split('-')[1]), 1));
      final endTs   = Timestamp.fromDate(
          DateTime(int.parse(monthStr.split('-')[0]),
                   int.parse(monthStr.split('-')[1]) + 1, 0, 23, 59));

      final advSnap = await _db.collection('payments')
          .where('labourId', isEqualTo: labourId)
          .where('type', isEqualTo: 'advance')
          .where('date', isGreaterThanOrEqualTo: startTs)
          .where('date', isLessThanOrEqualTo:    endTs)
          .get();

      final advances = advSnap.docs.fold(0.0,
          (s, d) => s + ((d.data()['amount'] as num?)?.toDouble() ?? 0));

      results.add(LabourReport(
        labourId:    labourId,
        labourName:  labData['name'] as String? ?? '',
        phone:       labData['phone'] as String? ?? '',
        dailyWage:   dailyWage,
        otRate:      otRate,
        daysPresent: daysPresent,
        daysHalf:    daysHalf,
        daysAbsent:  daysAbsent,
        totalOT:     totalOT,
        totalDays:   totalDays,
        basePay:     totalDays * dailyWage,
        otPay:       totalOT * otRate,
        grossSalary: gross,
        advances:    advances,
        netPayable:  gross - advances,
      ));
    }

    results.sort((a, b) => a.labourName.compareTo(b.labourName));
    reports       = results;
    totalGross    = results.fold(0, (s, r) => s + r.grossSalary);
    totalAdvances = results.fold(0, (s, r) => s + r.advances);
    totalNet      = results.fold(0, (s, r) => s + r.netPayable);
    isLoading     = false;
    error         = null;
    notifyListeners();
  }

  void previousMonth() {
    if (selectedMonth == 1) {
      selectedMonth = 12; selectedYear--;
    } else {
      selectedMonth--;
    }
    loadReport();
  }

  void nextMonth() {
    final now = DateTime.now();
    if (selectedYear < now.year ||
        (selectedYear == now.year && selectedMonth < now.month)) {
      if (selectedMonth == 12) {
        selectedMonth = 1; selectedYear++;
      } else {
        selectedMonth++;
      }
      loadReport();
    }
  }

  void _cancelSubs() {
    for (var s in _subs) s.cancel();
    _subs.clear();
  }

  @override
  void dispose() { _cancelSubs(); super.dispose(); }
}

// Data model

    required this.labourId,
    required this.labourName,
    required this.phone,
    required this.dailyWage,
    required this.otRate,
    required this.daysPresent,
    required this.daysHalf,
    required this.daysAbsent,
    required this.totalOT,
    required this.totalDays,
    required this.basePay,

  });
  final String labourId, labourName, phone;

}

// lib/providers/labour_provider.dart

class LabourProvider extends ChangeNotifier {
  final _db  = FirebaseFirestore.instance;
  final _auth = FirebaseAuth.instance;

  List<Labour>         labours         = [];
  List<Labour>         filteredLabours = [];
  Map<String, double>  advanceMap      = {}; // labourId → advance balance
  bool   isLoading  = false;
  String searchQuery = '';
  String? error;

  final List<StreamSubscription> _subs = [];
  String get uid => _auth.currentUser?.uid ?? '';

  Future<void> initialize() async {
    isLoading = true;
    notifyListeners();
    _startLabourStream();
    _startAdvanceStream();
  }

  void _startLabourStream() {
    final sub1 = _db.collection('labours')
        .where('supervisorId', isEqualTo: uid)
        .where('isActive', isEqualTo: true)
        .snapshots()
        .listen(_handleLabourSnap,
                onError: (e) => debugPrint('labour stream: $e'));
    _subs.add(sub1);

    final sub2 = _db.collection('labours')
        .where('contractorId', isEqualTo: uid)
        .where('isActive', isEqualTo: true)
        .snapshots()
        .listen(_handleLabourSnap,
                onError: (e) => debugPrint('labour contractorId stream: $e'));
    _subs.add(sub2);
  }

  final Map<String, Labour> _labourMap = {};

  void _handleLabourSnap(QuerySnapshot snap) {
    for (var d in snap.docs) {
      _labourMap[d.id] = Labour.fromFirestore(d);
    }
    labours         = _labourMap.values.toList()
      ..sort((a, b) => a.name.compareTo(b.name));
    isLoading       = false;
    _applySearch();
  }

  // Live advance balance per labour
  void _startAdvanceStream() {
    final sub = _db.collection('payments')
        .where('supervisorId', isEqualTo: uid)
        .where('type', isEqualTo: 'advance')
        .snapshots()
        .listen((snap) {
      final newMap = <String, double>{};
      for (var d in snap.docs) {
        final lid    = d.data()['labourId'] as String? ?? '';
        final amount = (d.data()['amount'] as num?)?.toDouble() ?? 0;
        newMap[lid]  = (newMap[lid] ?? 0) + amount;
      }
      advanceMap = newMap;
      notifyListeners();
    }, onError: (e) => debugPrint('advance stream: $e'));
    _subs.add(sub);
  }

  double advanceFor(String labourId) => advanceMap[labourId] ?? 0;

  void search(String query) {
    searchQuery = query;
    _applySearch();
    notifyListeners();
  }

  void _applySearch() {
    if (searchQuery.isEmpty) {
      filteredLabours = labours;
    } else {
      filteredLabours = labours.where((l) =>
        l.name.toLowerCase().contains(searchQuery.toLowerCase()) ||
        l.phone.contains(searchQuery)).toList();
    }
    notifyListeners();
  }

  Future<void> addLabour(Labour labour) async {
    try {
      final docRef = await _db.collection('labours').add({
        'supervisorId':        uid,
        'contractorId':        uid,
        'name':                labour.name,
        'phone':               labour.phone,
        'dailyWage':           labour.dailyWage,
        'overtimeWagePerHour': labour.overtimeWagePerHour,
        'skill':               labour.skill,
        'isActive':            true,
        'isSynced':            true,
        'syncedAt':            FieldValue.serverTimestamp(),
        'createdAt':           FieldValue.serverTimestamp(),
      });
      await docRef.update({'id': docRef.id});
      // Stream will auto-update the list — no manual rebuild needed
    } catch (e) {
      error = e.toString();
      notifyListeners();
    }
  }

  Future<void> updateLabour(Labour labour) async {
    if (labour.firestoreId == null) return;
    try {
      await _db.collection('labours').doc(labour.firestoreId).update({
        'name':                labour.name,
        'phone':               labour.phone,
        'dailyWage':           labour.dailyWage,
        'overtimeWagePerHour': labour.overtimeWagePerHour,
        'skill':               labour.skill,
        'updatedAt':           FieldValue.serverTimestamp(),
      });
      // Stream auto-updates
    } catch (e) {
      error = e.toString();
      notifyListeners();
    }
  }

  Future<void> deleteLabour(Labour labour) async {
    if (labour.firestoreId == null) return;
    try {
      await _db.collection('labours')
          .doc(labour.firestoreId)
          .update({'isActive': false, 'deletedAt': FieldValue.serverTimestamp()});
      _labourMap.remove(labour.id);
      labours = _labourMap.values.toList();
      _applySearch();
    } catch (e) {
      error = e.toString();
      notifyListeners();
    }
  }

  @override
  void dispose() {
    for (var s in _subs) s.cancel();
    super.dispose();
  }
}

export function useLabours(supervisorId = null) {
  const [labours, setLabours] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error,   setError]   = useState(null);

  useEffect(() => {
    const constraints = [where('isActive', '==', true)];
    if (supervisorId) {
      constraints.push(where('supervisorId', '==', supervisorId));
    }

    const q = query(collection(db, 'labours'), ...constraints);

    const unsub = onSnapshot(q,
      (snap) => {
        const data = snap.docs.map(d => ({
          id: d.id, ...d.data(),
          joiningDate: d.data().joiningDate?.toDate?.() || null,
        }));
        setLabours(data);
        setLoading(false);
      },
      (err) => {
        setError(err.message);
        setLoading(false);
      }
    );

    return () => unsub(); // cleanup on unmount
  }, [supervisorId]);

  return { labours, loading, error };
}
