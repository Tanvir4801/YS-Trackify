## Step 3 — Update ScannerService.processScan

Add processScanWithType method that accepts status parameter:

```dart
// Add this method to ScannerService:
Future<ScanResult> processScanWithType({
  required String rawToken,
  required String status, // 'present' | 'half'
}) async {
  // Decode token (same validation as before)
  final decoded = decodeQRToken(rawToken);
  if (decoded == null) {
    return ScanResult(
      success: false,
      message: 'Invalid QR Code',
      type: ScanResultType.invalid,
    );
  }
  if (decoded['error'] == 'expired') {
    return ScanResult(
      success: false,
      message: 'QR Code expired. Ask labour to refresh.',
      type: ScanResultType.expired,
    );
  }

  final labourId = decoded['labourId']!;

  // Check duplicate
  final alreadyMarked = await isAlreadyMarkedToday(labourId);
  if (alreadyMarked) {
    final name = await _getLabourName(labourId);
    return ScanResult(
      success: false,
      message: '$name already marked today',
      type: ScanResultType.duplicate,
      labourName: name,
    );
  }

  // Check connectivity and process
  final connectivity = await Connectivity().checkConnectivity();
  final isOnline = !connectivity.contains(ConnectivityResult.none);

  if (isOnline) {
    return await _markViaCloudFunction(labourId, rawToken, status);
  } else {
    return await _markOffline(labourId, status);
  }
}

// Update _markOffline to accept status:
Future<ScanResult> _markOffline(String labourId, String status) async {
  final today = _todayString();
  final name  = await _getLabourName(labourId);

  _saveToHive(labourId, status, isSynced: false);

  final statusLabel = status == 'half' ? 'Half Day' : 'Present';
  return ScanResult(
    success: true,
    message: '$name marked $statusLabel (offline). Will sync later.',
    type: ScanResultType.offline,
    labourName: name,
  );
}

// Update _markViaCloudFunction to accept status:
Future<ScanResult> _markViaCloudFunction(
    String labourId, String rawToken, String status) async {
  try {
    final callable =
        FirebaseFunctions.instance.httpsCallable('validateAndMarkAttendance');
    final result = await callable.call({
      'token':        rawToken,
      'supervisorId': supervisorId,
      'date':         _todayString(),
      'status':       status, // pass selected status
    });

    final data = result.data as Map<String, dynamic>;
    if (data['success'] == true) {
      final name        = data['labourName'] ?? 'Labour';
      final statusLabel = status == 'half' ? 'Half Day' : 'Present';
      _saveToHive(labourId, status, isSynced: true);
      return ScanResult(
        success: true,
        message: '$name marked $statusLabel ✓',
        type: ScanResultType.success,
        labourName: name,
      );
    } else {
      return ScanResult(
        success: false,
        message: data['message'] ?? 'Validation failed',
        type: ScanResultType.invalid,
      );
    }
  } on FirebaseFunctionsException catch (e) {
    if (e.code == 'already-exists') {
      return ScanResult(
        success: false,
        message: 'Already marked today',
        type: ScanResultType.duplicate,
      );
    }
    return await _markOffline(labourId, status);
  }
}
```

double calculateMonthlySalary({
  required int daysPresent,
  required int daysHalf,
  required double totalOTHours,
  required double dailyWage,
  required double otRate,
}) {
  // Regular pay
  final workedDays = daysPresent + (daysHalf * 0.5);
  final regularPay = workedDays * dailyWage;

  // Overtime pay
  final overtimePay = totalOTHours * otRate;

}
