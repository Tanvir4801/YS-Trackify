Future<int> syncPendingScans() async {
  final box = Hive.box('pending_attendance');
  final uid = FirebaseAuth.instance.currentUser?.uid ?? '';
  final contractorId = SessionService.instance.contractorId ?? uid;
  
  if (uid.isEmpty) {
    debugPrint('syncPendingScans: not logged in');
    return 0;
  }

  int synced = 0;
  final db = FirebaseFirestore.instance;

  for (var key in box.keys.toList()) {
    final rawRecord = box.get(key);
    if (rawRecord == null) continue;
    
    final record = Map<String, dynamic>.from(rawRecord as Map);
    
    // Skip already synced
    if (record['isSynced'] == true) continue;

    final labourId = record['labourId'] as String? ?? '';
    final date     = record['date']     as String? ?? '';
    final status   = record['status']   as String? ?? 'present';

    if (labourId.isEmpty || date.isEmpty) {
      // Invalid record — mark as done to remove from queue
      record['isSynced'] = true;
      await box.put(key, record);
      continue;
    }

    try {
      // Check if attendance already exists for this labour+date
      final existing = await db
          .collection('attendance')
          .where('labourId',     isEqualTo: labourId)
          .where('date',         isEqualTo: date)
          .where('supervisorId', isEqualTo: uid)
          .limit(1)
          .get();

      if (existing.docs.isNotEmpty) {
        // Already exists — just mark synced
        record['isSynced'] = true;
        await box.put(key, record);
        synced++;
        continue;
      }

      // Write to Firestore
      final docRef = await db.collection('attendance').add({
        'labourId':     labourId,
        'supervisorId': uid,
        'contractorId': contractorId,
        'date':         date,
        'status':       status,
        'overtimeHours': record['overtimeHours'] ?? 0,
        'markedVia':    'offline_qr',
        'isSynced':     true,
        'syncedAt':     FieldValue.serverTimestamp(),
      });
      await docRef.update({'id': docRef.id});

      record['isSynced']    = true;
      record['firestoreId'] = docRef.id;
      await box.put(key, record);
      synced++;
      
      debugPrint('Synced scan: $labourId → $status on $date');
    } catch (e) {
      debugPrint('Sync failed for $labourId: $e');
      // Don't mark as synced — will retry next time
    }
  }

  debugPrint('syncPendingScans complete: $synced synced');
  return synced;
}

int getPendingCount() {
  try {
    final box = Hive.box('pending_attendance');
    return box.values
        .where((v) {
          final m = v as Map;
          return m['isSynced'] != true;
        })
        .length;
  } catch (_) {
    return 0;
  }
}

// lib/core/utils/contractor_utils.dart

class ContractorUtils {
  
  // Returns true if this labour belongs to current supervisor
  static bool labourBelongsToSupervisor(
      Labour labour, String uid, String contractorId) {
    return labour.supervisorId == uid ||
           labour.contractorId == uid ||
           labour.contractorId == contractorId;
  }
  
  // Filter a list of labours for current supervisor only
  static List<Labour> filterForSupervisor(
      List<Labour> all, String uid, String contractorId) {
    return all.where((l) => 
      l.isActive && labourBelongsToSupervisor(l, uid, contractorId)
    ).toList();
  }
}
