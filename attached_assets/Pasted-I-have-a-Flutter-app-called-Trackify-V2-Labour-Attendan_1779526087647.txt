import 'dart:async';
import 'package:connectivity_plus/connectivity_plus.dart';
import 'package:flutter/material.dart';

class ConnectivityBanner extends StatefulWidget {
  const ConnectivityBanner({super.key, required this.child});
  final Widget child;

  @override
  State<ConnectivityBanner> createState() => _ConnectivityBannerState();
}

class _ConnectivityBannerState extends State<ConnectivityBanner>
    with SingleTickerProviderStateMixin {
  bool _isOnline = true;
  bool _justCameBack = false;
  StreamSubscription? _sub;
  late AnimationController _controller;
  late Animation<Offset> _slideAnim;
  Timer? _hideTimer;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 300),
    );
    _slideAnim = Tween<Offset>(
      begin: const Offset(0, -1),
      end: Offset.zero,
    ).animate(CurvedAnimation(
      parent: _controller, curve: Curves.easeOut));

    _checkInitial();
    _sub = Connectivity().onConnectivityChanged.listen((results) {
      final online = !results.contains(ConnectivityResult.none);
      if (online && !_isOnline) {
        // Just came back online
        setState(() { _isOnline = true; _justCameBack = true; });
        _controller.forward();
        _hideTimer?.cancel();
        _hideTimer = Timer(const Duration(seconds: 3), () {
          if (mounted) {
            _controller.reverse().then((_) {
              if (mounted) setState(() => _justCameBack = false);
            });
          }
        });
      } else if (!online && _isOnline) {
        setState(() { _isOnline = false; _justCameBack = false; });
        _controller.forward();
      }
    });
  }

  Future<void> _checkInitial() async {
    final result = await Connectivity().checkConnectivity();
    final online = !result.contains(ConnectivityResult.none);
    if (!online && mounted) {
      setState(() => _isOnline = false);
      _controller.forward();
    }
  }

  @override
  void dispose() {
    _sub?.cancel();
    _controller.dispose();
    _hideTimer?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        // Banner
        SlideTransition(
          position: _slideAnim,
          child: AnimatedSwitcher(
            duration: const Duration(milliseconds: 200),
            child: _isOnline && _justCameBack
                ? _buildBanner(
                    color: const Color(0xFF16A34A),
                    icon: Icons.cloud_done_rounded,
                    message: 'Back online — syncing data...',
                    key: const ValueKey('online'),
                  )
                : !_isOnline
                    ? _buildBanner(
                        color: const Color(0xFFD97706),
                        icon: Icons.cloud_off_rounded,
                        message: 'Offline — scans are queued',
                        key: const ValueKey('offline'),
                      )
                    : const SizedBox.shrink(key: ValueKey('none')),
          ),
        ),
        // Rest of app
        Expanded(child: widget.child),
      ],
    );
  }

  Widget _buildBanner({
    required Color color,
    required IconData icon,
    required String message,
    required Key key,
  }) {
    return Container(
      key: key,
      width: double.infinity,
      color: color,
      padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
      child: Row(children: [
        Icon(icon, color: Colors.white, size: 16),
        const SizedBox(width: 8),
        Text(message,
          style: const TextStyle(
            color: Colors.white,
            fontSize: 12,
            fontWeight: FontWeight.w600,
          ),
        ),
        const Spacer(),
        if (!_isOnline)
          const SizedBox(
            width: 12, height: 12,
            child: CircularProgressIndicator(
              strokeWidth: 2,
              valueColor: AlwaysStoppedAnimation(Colors.white),
            ),
          ),
      ]),
    );
  }
}

Future<void> syncPendingAttendance() async {
  final uid = FirebaseAuth.instance.currentUser?.uid ?? '';
  if (uid.isEmpty) return;

  final box = Hive.box<Attendance>(Attendance.boxName);
  final pending = box.values
      .where((a) => !a.isSynced && a.supervisorId == uid)
      .toList();

  if (pending.isEmpty) return;

  debugPrint('Auto-syncing ${pending.length} attendance records...');
  int synced = 0;

  for (final att in pending) {
    try {
      final existing = await FirebaseFirestore.instance
          .collection('attendance')
          .where('labourId',     isEqualTo: att.labourId)
          .where('supervisorId', isEqualTo: uid)
          .where('date',         isEqualTo: att.date)
          .limit(1).get();

      if (existing.docs.isNotEmpty) {
        await existing.docs.first.reference.update({
          'status':        att.status.firestoreValue,
          'overtimeHours': att.overtimeHours,
          'isSynced':      true,
        });
      } else {
        final ref = await FirebaseFirestore.instance
            .collection('attendance').add({
          'labourId':      att.labourId,
          'supervisorId':  uid,
          'contractorId':  SessionService.instance.contractorId ?? uid,
          'date':          att.date,
          'status':        att.status.firestoreValue,
          'overtimeHours': att.overtimeHours,
          'markedVia':     'manual_offline',
          'isSynced':      true,
          'syncedAt':      FieldValue.serverTimestamp(),
        });
        await ref.update({'id': ref.id});
      }
      att.isSynced = true;
      await att.save();
      synced++;
    } catch (e) {
      debugPrint('Sync att error: $e');
    }
  }
  debugPrint('Attendance sync complete: $synced');
}

import 'package:flutter/material.dart';
import 'package:local_auth/local_auth.dart';
import 'package:shared_preferences/shared_preferences.dart';

class AppLockService {
  static final instance = AppLockService._();
  AppLockService._();

  final _auth = LocalAuthentication();

  // Check if device supports biometrics
  Future<bool> canUseBiometrics() async {
    try {
      return await _auth.canCheckBiometrics &&
             await _auth.isDeviceSupported();
    } catch (_) { return false; }
  }

  // Authenticate with biometrics or PIN
  Future<bool> authenticate() async {
    try {
      return await _auth.authenticate(
        localizedReason: 'Verify your identity to open Trackify',
        options: const AuthenticationOptions(
          biometricOnly: false, // allows PIN fallback
          stickyAuth:    true,
        ),
      );
    } catch (e) {
      debugPrint('Biometric auth error: $e');
      return false;
    }
  }

  // Check if app lock is enabled
  Future<bool> isLockEnabled() async {
    final prefs = await SharedPreferences.getInstance();
    return prefs.getBool('app_lock_enabled') ?? false;
  }

  Future<void> setLockEnabled(bool enabled) async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setBool('app_lock_enabled', enabled);
  }
}

import 'package:flutter_local_notifications/flutter_local_notifications.dart';
import 'package:timezone/timezone.dart' as tz;
import 'package:timezone/data/latest.dart' as tzData;

class NotificationService {
  static final instance = NotificationService._();
  NotificationService._();

  final _plugin = FlutterLocalNotificationsPlugin();
  bool _initialized = false;

  Future<void> initialize() async {
    if (_initialized) return;
    tzData.initializeTimeZones();

    const androidSettings = AndroidInitializationSettings(
      '@mipmap/ic_launcher');
    const iosSettings = DarwinInitializationSettings(
      requestAlertPermission:  true,
      requestBadgePermission:  true,
      requestSoundPermission:  true,
    );

    await _plugin.initialize(
      const InitializationSettings(
        android: androidSettings, iOS: iosSettings),
    );
    _initialized = true;
  }

  // Notify supervisor: attendance not marked for today
  Future<void> scheduleAttendanceReminder() async {
    await initialize();

    final now  = DateTime.now();
    final nine = DateTime(now.year, now.month, now.day, 9, 0);
    var scheduleTime = nine;
    if (now.isAfter(nine)) {
      scheduleTime = nine.add(const Duration(days: 1));
    }

    await _plugin.zonedSchedule(
      1,
      '⏰ Mark Attendance',
      'Don\'t forget to mark attendance for today!',
      tz.TZDateTime.from(scheduleTime, tz.local),
      const NotificationDetails(
        android: AndroidNotificationDetails(
          'attendance_reminder',
          'Attendance Reminder',
          channelDescription: 'Daily attendance reminder',
          importance: Importance.high,
          priority:   Priority.high,
          icon:       '@mipmap/ic_launcher',
        ),
        iOS: DarwinNotificationDetails(),
      ),
      androidScheduleMode: AndroidScheduleMode.exactAllowWhileIdle,
      matchDateTimeComponents: DateTimeComponents.time,
      uiLocalNotificationDateInterpretation:
          UILocalNotificationDateInterpretation.absoluteTime,
    );
  }

  // Instant notification: scan successful
  Future<void> showScanSuccess(String labourName, String status) async {
    await initialize();
    final statusLabel = status == 'half' ? 'Half Day' : 'Present';
    await _plugin.show(
      DateTime.now().millisecondsSinceEpoch ~/ 1000,
      '✅ Attendance Marked',
      '$labourName marked $statusLabel',
      const NotificationDetails(
        android: AndroidNotificationDetails(
          'scan_success', 'Scan Success',
          channelDescription: 'QR scan confirmations',
          importance: Importance.defaultImportance,
          priority:   Priority.defaultPriority,
          icon:       '@mipmap/ic_launcher',
        ),
        iOS: DarwinNotificationDetails(),
      ),
    );
  }

  // Sync complete notification
  Future<void> showSyncComplete(int count) async {
    if (count == 0) return;
    await initialize();
    await _plugin.show(
      999,
      '☁️ Sync Complete',
      '$count attendance records synced to cloud',
      const NotificationDetails(
        android: AndroidNotificationDetails(
          'sync_complete', 'Sync Complete',
          channelDescription: 'Data sync notifications',
          importance: Importance.low,
          priority:   Priority.low,
          icon:       '@mipmap/ic_launcher',
        ),
        iOS: DarwinNotificationDetails(),
      ),
    );
  }

  Future<void> cancelAll() async {
    await initialize();
    await _plugin.cancelAll();
  }
}

// lib/widgets/week_attendance_strip.dart

class WeekAttendanceStrip extends StatelessWidget {
  const WeekAttendanceStrip({
    super.key,
    required this.attendanceByDate,
    // Map<dateString, Map<status, count>>
    // e.g. {'2026-05-01': {'present': 8, 'absent': 2, 'half': 1}}
  });

  final Map<String, Map<String, int>> attendanceByDate;

  @override
  Widget build(BuildContext context) {
    final days = List.generate(7, (i) {
      final d = DateTime.now().subtract(Duration(days: 6 - i));
      return d;
    });

    return Row(
      children: days.map((day) {
        final key = '${day.year}-'
            '${day.month.toString().padLeft(2,'0')}-'
            '${day.day.toString().padLeft(2,'0')}';
        final data    = attendanceByDate[key];
        final present = data?['present'] ?? 0;
        final absent  = data?['absent']  ?? 0;
        final half    = data?['half']    ?? 0;
        final total   = present + absent + half;
        final isToday = key == _todayKey();

        return Expanded(
          child: Column(children: [
            Text(
              _dayLabel(day),
              style: TextStyle(
                fontSize: 10,
                fontWeight: isToday
                    ? FontWeight.w700 : FontWeight.w400,
                color: isToday
                    ? const Color(0xFF1B8B6E) : Colors.grey,
              ),
            ),
            const SizedBox(height: 4),
            Container(
              height: 36,
              margin: const EdgeInsets.symmetric(horizontal: 2),
              decoration: BoxDecoration(
                borderRadius: BorderRadius.circular(6),
                color: total == 0
                    ? Colors.grey.shade100
                    : null,
                gradient: total > 0 ? LinearGradient(
                  begin: Alignment.bottomCenter,
                  end:   Alignment.topCenter,
                  colors: [
                    Colors.green.withValues(
                      alpha: (present / total).clamp(0.2, 1.0)),
                    Colors.green.shade300,
                  ],
                ) : null,
                border: isToday ? Border.all(
                  color: const Color(0xFF1B8B6E), width: 2) : null,
              ),
              child: Center(
                child: Text(
                  total > 0 ? '$present' : '-',
                  style: TextStyle(
                    fontSize: 11,
                    fontWeight: FontWeight.w700,
                    color: total > 0
                        ? Colors.white : Colors.grey.shade400,
                  ),
                ),
              ),
            ),
            const SizedBox(height: 4),
            Text(
              '${day.day}',
              style: TextStyle(
                fontSize: 9,
                color: isToday
                    ? const Color(0xFF1B8B6E) : Colors.grey,
              ),
            ),
          ]),
        );
      }).toList(),
    );
  }

  String _dayLabel(DateTime d) {
    const days = ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'];
    return days[d.weekday - 1];
  }

  String _todayKey() {
    final n = DateTime.now();
    return '${n.year}-'
        '${n.month.toString().padLeft(2,'0')}-'
        '${n.day.toString().padLeft(2,'0')}';
  }
}

Map<String, Map<String, int>> weekAttendance = {};

Future<void> _loadWeekAttendance(String uid) async {
  final now   = DateTime.now();
  final start = now.subtract(const Duration(days: 6));
  final startStr = '${start.year}-'
      '${start.month.toString().padLeft(2,'0')}-'
      '${start.day.toString().padLeft(2,'0')}';

  try {
    final snap = await FirebaseFirestore.instance
        .collection('attendance')
        .where('supervisorId', isEqualTo: uid)
        .where('date', isGreaterThanOrEqualTo: startStr)
        .get();

    final Map<String, Map<String, int>> result = {};
    for (var doc in snap.docs) {
      final date   = doc.data()['date']   as String? ?? '';
      final status = doc.data()['status'] as String? ?? '';
      if (date.isEmpty) continue;
      result[date] ??= {'present': 0, 'absent': 0, 'half': 0};
      result[date]![status] = (result[date]![status] ?? 0) + 1;
    }
    weekAttendance = result;
    notifyListeners();
  } catch (e) {
    debugPrint('_loadWeekAttendance error: $e');
  }
}
